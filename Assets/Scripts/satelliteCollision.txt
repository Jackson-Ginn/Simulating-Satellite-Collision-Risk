using UnityEngine;

public static class satelliteCollision
{
    private struct TCA
    {
        public float time;
        public Vector3 pos;
    }
    public static float collisionRisk(
        float[,,] uncA, float[,,] uncB,
        Vector3 posA, Vector3 posB,
        Vector3 velA, Vector3 velB,
        float sizeA, float sizeB)
    {
        //float TCA = TimeOfClosestApproach(posA, velA, posB, velB);
        float[,,] combinedUncertainty = CombineUncertainties(uncA, uncB);
        float collisionRadius = sizeA + sizeB;
    }

    public static TCA TimeOfClosestApproach(
        Vector3 posA, Vector3 velA,
        Vector3 posB, Vector3 velB)
    {
        Vector3 relPos = posB - posA;
        Vector3 relVel = velB - velA;

        float relSpeedSq = relVel.sqrMagnitude;
        if (relSpeedSq == 0) return 0; // No relative motion

        return new TCA
        {
            time = -Vector3.Dot(relPos, relVel) / relSpeedSq,
            pos = posA + velA * (-Vector3.Dot(relPos, relVel) / relSpeedSq)
        };
    }
}
